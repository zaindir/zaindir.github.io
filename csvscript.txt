Set-Location "C:\xampp\htdocs\cgi\bin\assets\files"

Get-ChildItem -File | Select-Object `
@{Name="Nama Game"; Expression = { $_.BaseName }},
@{Name="Ukuran"; Expression = {
    if ($_.Length -ge 1MB) {
        "{0:N2} MB" -f ($_.Length / 1MB)
    } else {
        "{0:N0} KB" -f ($_.Length / 1KB)
    }
}},
@{Name="Lokasi"; Expression = {
    "cgi/bin/assets/files/" + $_.Name
}} |
Export-Csv -Path "C:\xampp\htdocs\data.csv" -NoTypeInformation -Encoding UTF8
